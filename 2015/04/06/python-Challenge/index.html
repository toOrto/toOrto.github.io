new_list = [0] * len(orig)
final = ""
for i in xrange(0,len(orig)):
    new_list[i] += ord(orig[i])
    if new_list[i] == 0x20:
        final += " "
        continue
    if new_list[i] == 0x2e:
        final += "."
        continue
    if new_list[i] == 0x27:
        final += "'"
        continue
    if new_list[i] == 0x28:
        final += "("
        continue
    if new_list[i] == 0x29:
        final += ")"
        continue
    new_list[i] = new_list[i]+2
    if new_list[i] > 0x7a:
        new_list[i] = 0x61 + (new_list[i] - 0x7a -1)
    final += chr(new_list[i])
print new_list
print "\n"+final + "\n"
